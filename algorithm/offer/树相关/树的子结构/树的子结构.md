        public boolean isSubtree(TreeNode s, TreeNode t) {
            if(s == null || t == null) {
                return false;
            }
            if(isSameTree(s, t)) {
                return true;
            }
            return isSubtree(s.left, t) || isSubtree(s.right, t);
        }
    
        private boolean isSameTree(TreeNode s, TreeNode t) {
            if(s == null) {
                return t == null;
            }
            if(t == null) {
                return false;
            }
            if(s.val != t.val) {
                return false;
            }
            return isSameTree(s.left, t.left) && isSameTree(s.right, t.right);
        }
